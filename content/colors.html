All basic drawing functions in P5.js allows you to specify the size of the shape you're drawing. To draw a rectangle, we use the rect(x, y, width, height) function. In the following example, we draw a rectangle that takes up half the width of canvas.

function setup() {
  createCanvas(600, 450);
  background(255, 255, 220);

  if(myNum > 350) {
    console.log(myNum)
  }

  rectMode(CENTER);
  noStroke();
  fill(40);
  translate(width * 0.2, height * 0.56);
  rotate(radians(10));
  rect(0, 0, height * 0.4, height * 0.8);
  noLoop();
}
